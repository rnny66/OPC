```sql
-- Organizer invitations table
CREATE TABLE IF NOT EXISTS organizer_invitations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  email text NOT NULL,
  invited_by uuid REFERENCES profiles(id) ON DELETE SET NULL,
  created_at timestamptz DEFAULT now(),
  accepted_at timestamptz DEFAULT NULL,
  CONSTRAINT unique_invitation_email UNIQUE (email)
);

-- Enable RLS
ALTER TABLE organizer_invitations ENABLE ROW LEVEL SECURITY;

-- RLS policies: admin only
CREATE POLICY "Admins can read invitations"
  ON organizer_invitations FOR SELECT
  USING (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  );

CREATE POLICY "Admins can insert invitations"
  ON organizer_invitations FOR INSERT
  WITH CHECK (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  );

CREATE POLICY "Admins can update invitations"
  ON organizer_invitations FOR UPDATE
  USING (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  );

CREATE POLICY "Admins can delete invitations"
  ON organizer_invitations FOR DELETE
  USING (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  );

-- Trigger: auto-promote on signup if invited
CREATE OR REPLACE FUNCTION handle_organizer_invitation()
RETURNS trigger AS $$
BEGIN
  -- Check if this new user's email has a pending invitation
  UPDATE organizer_invitations
  SET accepted_at = now()
  WHERE email = NEW.email AND accepted_at IS NULL;

  IF FOUND THEN
    -- Promote the user to organizer
    NEW.role := 'organizer';
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Fire BEFORE the existing profile insert trigger so role is set on creation
CREATE TRIGGER check_organizer_invitation
  BEFORE INSERT ON profiles
  FOR EACH ROW
  EXECUTE FUNCTION handle_organizer_invitation();

-- Allow admins to update role on profiles (for promoting existing users)
CREATE POLICY "Admins can update user roles"
  ON profiles FOR UPDATE
  USING (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  )
  WITH CHECK (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')
  );
```

```ts
import { createSupabaseAdmin } from '@/lib/supabase/admin'

export async function promoteToOrganizer(userId: string) {
  await requireAdmin()
  const adminClient = createSupabaseAdmin()

  const { error } = await adminClient
    .from('profiles')
    .update({ role: 'organizer' })
    .eq('id', userId)

  if (error) return { error: error.message }

  revalidatePath('/admin/users')
  return { success: true }
}

export async function inviteOrganizer(email: string) {
  await requireAdmin()
  const adminClient = createSupabaseAdmin()

  const { error } = await adminClient
    .from('organizer_invitations')
    .insert({ email, invited_by: (await (await createSupabaseServer()).auth.getUser()).data.user?.id })

  if (error) return { error: error.message }

  revalidatePath('/admin/organizers')
  return { success: true }
}

export async function cancelTournamentAdmin(tournamentId: string) {
  await requireAdmin()
  const adminClient = createSupabaseAdmin()

  const { error } = await adminClient
    .from('tournaments')
    .update({ status: 'cancelled' })
    .eq('id', tournamentId)

  if (error) return { error: error.message }

  revalidatePath('/admin/tournaments')
  return { success: true }
}
```

Note: The `import { createSupabaseAdmin }` must be added at the top of the file alongside existing imports.

```tsx
export function InviteOrganizerForm() {
  const [sending, setSending] = useState(false)
  const [message, setMessage] = useState<{ type: 'success' | 'error'; text: string } | null>(null)

  async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()
    setSending(true)
    setMessage(null)

    const formData = new FormData(e.currentTarget)
    const email = formData.get('email') as string

    const result = await inviteOrganizer(email)

    if (result.error) {
      setMessage({ type: 'error', text: result.error })
    } else {
      setMessage({ type: 'success', text: `Invitation sent for ${email}` })
      e.currentTarget.reset()
    }
    setSending(false)
  }

  return (
    <div>
      <form onSubmit={handleSubmit} style={formStyles.form}>
        <input
          name="email"
          type="email"
          required
          placeholder="organizer@example.com"
          style={formStyles.input}
        />
        <button type="submit" disabled={sending} style={formStyles.button}>
          {sending ? 'Sending...' : 'Send Invite'}
        </button>
      </form>
      {message && (
        <p style={{
          ...formStyles.message,
          color: message.type === 'success' ? '#22c55e' : '#ef4444',
        }}>
          {message.text}
        </p>
      )}
    </div>
  )
}
```

```tsx
    registration_count: t.tournament_registrations?.[0]?.count || 0,
  }))

  return (
    <div>
      <h1 style={styles.title}>Tournament Oversight</h1>
      <p style={styles.subtitle}>View and manage all tournaments across organizers</p>

      <form style={styles.filterBar}>
        <select name="status" defaultValue={statusFilter} style={styles.select}>
          <option value="">All Statuses</option>
          <option value="upcoming">Upcoming</option>
          <option value="ongoing">Ongoing</option>
          <option value="completed">Completed</option>
          <option value="cancelled">Cancelled</option>
        </select>
        <select name="organizer" defaultValue={organizerFilter} style={styles.select}>
          <option value="">All Organizers</option>
          {(organizerList || []).map((org: any) => (
            <option key={org.id} value={org.id}>{org.display_name || org.id}</option>
          ))}
        </select>
        <button type="submit" style={styles.button}>Filter</button>
      </form>

      <AdminTournamentTable tournaments={tableData} />
    </div>
  )
}
```
